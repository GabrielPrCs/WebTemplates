Buttons

Basic Buttons

            
              // Default button
              <button class="button default-button">Default</button>

              // Primary button
              <button class="button primary-button">Primary</button>

              // Danger button
              <button class="button danger-button">Danger</button>

              // Success button
              <button class="button success-button">Success</button>

              // Info button
              <button class="button info-button">Info</button>

              // Warning button
              <button class="button warning-button">Warning</button>

              // Disabled button (works with all the buttons above)
              <button class="button warning-button" disabled>Disabled</button>

Outline Buttons

            
              // Default outline button
              <button class="button default-button outline-button">Default Outline</button>

              // Primary outline button
              <button class="button primary-button outline-button">Primary Outline</button>

              // Danger outline button
              <button class="button danger-button outline-button">Danger Outline</button>

              // Success outline button
              <button class="button success-button outline-button">Success Outline</button>

              // Info outline button
              <button class="button info-button outline-button">Info Outline</button>

              // Warning outline button
              <button class="button warning-button outline-button">Warning Outline</button>

              // Disabled outline button (works with all the buttons above)
              <button class="button warning-button outline-button" disabled>Disabled Outline</button>

Rounded Buttons (Works with any kind of button)

            
              <button class="button primary-button rounded-button">Rounded Button</button>
              <button class="button primary-button outline-button rounded-button">Rounded Outline Button</button>

Sizes (Works with any kind of button)

            
              // Small button
              <button class="button primary-button small-button">Small Button</button>

              // Button
              <button class="button primary-button">Button</button>

              // Large button
              <button class="button primary-button large-button">Large Button</button>

              // Small block button
              <button class="button primary-button small-button block-button">Small Block Button</button>

              // Block button
              <button class="button primary-button block-button">Block Button</button>

              // Large block button
              <button class="button primary-button small-button block-button">Large Block Button</button>
