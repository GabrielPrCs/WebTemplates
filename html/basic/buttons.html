Buttons

Basic Buttons

            
              // Default button
              <button class="button default-button">Default Button</button>

              // Primary button
              <button class="button primary-button">Primary Button</button>

              // Danger button
              <button class="button danger-button">Danger Button</button>

              // Success button
              <button class="button success-button">Success Button</button>

              // Info button
              <button class="button info-button">Info Button</button>

              // Warning button
              <button class="button warning-button">Warning Button</button>

              // Disabled button (works with all the buttons above)
              <button class="button warning-button" disabled>Disabled Button</button>

Sizes

            
              // Small button
              <button class="button primary-button small-button">Small Button</button>

              // Button
              <button class="button primary-button">Button</button>

              // Large button
              <button class="button primary-button large-button">Large Button</button>

              // Small block button
              <button class="button primary-button small-button block-button">Small Block Button</button>

              // Block button
              <button class="button primary-button block-button">Block Button</button>

              // Large block button
              <button class="button primary-button small-button block-button">Large Block Button</button>
